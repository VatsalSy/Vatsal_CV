\section{\textbf{Research Involvements}}
\tlcventry{2016}{0}{\large \textbf{Collision of liquid jets}}{}{}{}{}
\cvitem{Mentor}{Prof. Arup Kumar Das}
\cvitem{Description}{Numerical simulations are used to understand the fundamental Physics for different regimes observed as a result of collision of two liquid jets. Experiments are also conducted to explore the asymetries induced in the process and to fomulate the numerical model for simulations.}
\tlcventry{2015}{0}{\large \textbf{Bubble Entrainment by plunging liquid jets on pool}}{}{}{}{}
\cvitem{Metor}{Prof. Arup Kumar Das}
\cvitem{Description}{A series of controlled experiments and numerical simulations is carried out to delve into the process of bubble entrainment by plunging liquid jet. The initiation process is captured. Entrainment clusters obtained are throughly characterized and their interactions in case of multiple jet impinging onto the pool are studied.}
%\vspace{0.4cm}
\tlcventry{2015}{0}{\large \textbf{Multi-Scale Simulation}}{}{}{}{}
\cvitem{Mentor}{Prof. Arup Kumar Das}
\cvitem{Description}{Volume of Fluid (VOF) - Lagrangian Point Particle (LPP) based hybrid approach is used to study diffrent multi-scale phenomena, such as liquid jet atomization, bursting of bubble at the liquid pool surface and transition of two phase pipe flow from annular to droplet flow regime.}
%\vspace{0.15cm}
%\vspace{0.15cm}


\tlcventry{2015}{0}{\large \textbf{Numerical Simulation of Drainage of Kitchen Sink}}{}{}{}{}
\cvitem{Mentor}{Prof. Arup Kumar Das}
\cvitem{Description}{Three dimensional simulation of emptying of the kitchen sink is carried out to observe the vortex-like structures formed during the drainage. Dependence of the structures formed on fluid properties and geometry and placement of the drain is explored.}



\tlcventry{2015}{0}{\large \textbf{Numerical Simulation of phase change processes}}{}{}{}{}
\cvitem{Mentor}{Prof. Arup Kumar Das}
\cvitem{Description}{A coupled Volume of Fluid (VOF) - Phase change module is used to simulate the boiling process over a series of cylinders arranged in different configurations, such as vertical and horizontal inline, and staggered.}
%\vspace{0.15cm}



\tlcventry{2016}{2016}{\large \textbf{Numerical Investigation of the Landau film entrainment}}{}{}{}{}
\cvitem{Mentors}{Prof. J. John Soundar, Prof. Jean-Philippe Matas, Prof. Micka\"el Bourgoin}
\cvitem{Description}{The classical Landau film problem of the coating of a plate with liquid as the plate is dragged out of the liquid pool has been studied under this project. The model is first validated with Landau's expression for steady state film thickness, and then is used to characterize the overlap region between the film and the meniscus near the pool.}
%\vspace{0.15cm}


\tlcventry{2016}{2016}{\large \textbf{Numerical Investigation of Splashing from a Rotating Body}}{}{}{}{}
\cvitem{Mentors}{Prof. J. John Soundar, Prof. Jean-Philippe Matas, Prof. Micka\"el Bourgoin}
\cvitem{Description}{The aim of this project is to predict water dispersion around the wheel of a car crossing a ford. We have used a finite volume based computational multi-fluid dynamics solver, Gerris to investigate the phenomenon numerically.}


\tlcventry{2014}{0}{\large \textbf{Fire Dynamics}}{}{}{}{}
\cvitem{Mentor}{Prof. Arup Kumar Das}
\cvitem{Description}{The main aim of this project is the simulation of different fire scenarios, such as compartment fires and forest fires. Fire Dynamics Simulator (FDS) is used to study the propagation of fires and smoke flow pattern in enclosures such as buildings and railway coaches. Modelling of forest is done using obstructions of various sizes, subjected to fire. Scaled down experimental setup is also used to visualize the process.}
